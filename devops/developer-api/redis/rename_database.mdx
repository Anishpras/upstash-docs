---
title: "Rename Database"
description: "This endpoint renames a database."
api: "POST https://api.upstash.com/v2/redis/rename/{id}"
authMethod: "basic"
---

## URL Parameters

<ParamField path="id" type="string" required>
  The ID of the database to be renamed
</ParamField>

## Request Parameters

<ParamField body="name" type="string" required>
  The new name of the database
</ParamField>

## Response Parameters

<ResponseField name="database_id" type="string">
  ID of the created database
</ResponseField>
<ResponseField name="database_name" type="string">
  New name of the database
</ResponseField>
<ResponseField name="database_type" type="string">
  Type of the database in terms of pricing model\ `Free`, `Pay as You Go` or
  `Enterprise`
</ResponseField>
<ResponseField name="region" type="string">
  The region where database is hosted
</ResponseField>
<ResponseField name="primary_members" type="array<string>">
  List of primary regions in the database cluster
</ResponseField>
<ResponseField name="all_members" type="array<string>">
  List of all regions in the database cluster
</ResponseField>
<ResponseField name="primary_region" type="string">
  Primary region of the database cluster
</ResponseField>
<ResponseField name="port" type="int">
  Database port for clients to connect
</ResponseField>
<ResponseField name="creation_time" type="int">
  Creation time of the database as Unix time
</ResponseField>
<ResponseField name="budget" type="int">
  Allocated budget for database operations
</ResponseField>
<ResponseField name="state" type="string">
  State of database\ `active` or `deleted`
</ResponseField>
<ResponseField name="password" type="string">
  Password of the database
</ResponseField>
<ResponseField name="user_email" type="string">
  Email or team id of the owner of the database
</ResponseField>
<ResponseField name="endpoint" type="string">
  Endpoint URL of the database
</ResponseField>
<ResponseField name="tls" type="boolean">
  TLS/SSL is enabled or not
</ResponseField>
<ResponseField name="eviction" type="boolean">
  Whether entry eviction is enabled
</ResponseField>
<ResponseField name="auto_upgrade" type="boolean">
  Automatic upgrade capability status
</ResponseField>
<ResponseField name="consistent" type="boolean">
  Strong consistency mode status
</ResponseField>

<ResponseField name="reserved_per_region_price" type="int">
  Reserved price per region for enterprise plans
</ResponseField>

<ResponseField name="next_daily_backup_time" type="int">
  Unix timestamp of next scheduled backup
</ResponseField>

<ResponseField name="modifying_state" type="string">
  Database state 
</ResponseField>

<ResponseField name="rest_token" type="string">
  Full-access REST token
</ResponseField>

<ResponseField name="read_only_rest_token" type="string">
  Read-only REST token
</ResponseField>

<ResponseField name="db_max_clients" type="int">
  Maximum allowed concurrent client connections
</ResponseField>

<ResponseField name="db_max_request_size" type="int">
  Maximum request size in bytes
</ResponseField>

<ResponseField name="db_resource_size" type="string">
  Resource allocation tier
</ResponseField>

<ResponseField name="db_type" type="string">
  Database storage engine type
</ResponseField>

<ResponseField name="db_disk_threshold" type="int">
  Disk storage limit in bytes
</ResponseField>

<ResponseField name="db_max_entry_size" type="int">
  Maximum entry size in bytes
</ResponseField>

<ResponseField name="db_memory_threshold" type="int">
  Memory usage limit in bytes
</ResponseField>

<ResponseField name="db_conn_idle_timeout" type="int">
  Connection idle timeout in nanoseconds
</ResponseField>

<ResponseField name="db_lua_timeout" type="int">
  Lua script execution timeout in nanoseconds
</ResponseField>

<ResponseField name="db_lua_credits_per_min" type="int">

</ResponseField>

<ResponseField name="db_store_max_idle" type="int">

</ResponseField>

<ResponseField name="db_max_commands_per_second" type="int">

<ResponseField name="db_request_limit" type="int">

  Database-level eviction policy status

<ResponseField name="db_acl_enabled" type="boolean">
  Access Control List enablement status

<RequestExample>

```shell curl
curl -X POST \
  https://api.upstash.com/v2/redis/rename/:id \
  -u 'EMAIL:API_KEY'
  -d '{"name":"MyRedis_new_name"}'
```

```python Python
import requests

data = '{"name":"MyRedis_new_name"}'
response = requests.post('https://api.upstash.com/v2/redis/rename/:id', data=data, auth=('EMAIL', 'API_KEY'))
response.content
```

```go Go
client := &http.Client{}
var data = strings.NewReader(`{
    "name":"MyRedis_new_name"
}`)
req, err := http.NewRequest("POST", "https://api.upstash.com/v2/redis/rename/:id", data)
if err != nil {
    log.Fatal(err)
}
req.SetBasicAuth("email", "api_key")
resp, err := client.Do(req)
if err != nil {
    log.Fatal(err)
}
bodyText, err := ioutil.ReadAll(resp.Body)
if err != nil {
    log.Fatal(err)
}
fmt.Printf("%s\n", bodyText);
```
